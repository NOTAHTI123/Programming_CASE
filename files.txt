deployment.yml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: react-app
  labels:
    app: react-app
spec:
  replicas: 2
  selector:
    matchLabels:
      app: react-app
  template:
    metadata:
      labels:
        app: react-app
    spec:
      containers:
      - name: react-app
        image: kparwaz74/getting-started:jenkins-python-v1
        imagePullPolicy: Always
        ports:
        - containerPort: 5173
        env:
        - name: DB-PORT
          valueFrom:
            configMapKeyRef:
              name: sample-python-app-cm
              key: db-port


service.yml

apiVersion: v1
kind: Service
metadata:
  name: react-app-service
spec:
  type: NodePort
  selector:
    app: react-app
  ports:
    - protocol: TCP
      port: 80
      targetPort: 5173
      nodePort: 30001
