#include <iostream>
using namespace std;


int count(int n){
    if(n>=0&&n<=9){
        return 1;
    }
    else{
        n = n / 10;
        return 1 + count(n);
    }
}

int main(){
    int n;
    cout << "Enter the number : ";
    cin >> n;
    cout << endl;
    cout << "The number of digits are : " << count(n) << endl;
}
