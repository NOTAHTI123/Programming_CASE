#include <iostream>
using namespace std;

int fibo(int x)
{
    
   if((x==1)||(x==0)) {
      return(x);
   }else {
      return(fibo(x-1)+fibo(x-2));
   }
}

int main()
{
    int num;
    cout << "Enter the max value of fibonacci : ";
    cin >> num;
    cout << endl;
    int i = 0;
    while(i<num){
      cout << fibo(i);
      i++;
    }
}
