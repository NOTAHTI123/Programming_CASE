
#include <iostream>
using namespace std;

void binary_search(int arr[], int size, int to_find)
{
    int low = 0;
    int high = size;
    int half = (low + high) / 2;
    if (arr[half] == to_find)
    {
        cout << "Data found at index number : " << half << endl;
    }
    else
    {
        while (low != high)
        {
            half = (low + high) / 2;
            if (to_find < arr[half])
            {
                high = half-1;
            }
            else if (to_find > arr[half])
            {
                low = half + 1;
            }
            else if (arr[half] == to_find)
            {

                cout << "Data found at index number : " << half << endl;
                break;
            }
        }
    }
}

void display(int arr[], int size)
{
    int i = 0;
    while (i < size)
    {
        cout << arr[i] << "\t";

        i++;
    }
    cout << endl;
}

int main()
{

    int arr[8] = {2, 5, 52, 55, 68, 76, 85, 98};

    int size = sizeof(arr)/sizeof(arr[0]);

    int to_find;

    cout << "Array : ";
    display(arr, size);

    cout << "\n";

    cout << "Enter the number you want to find : ";
    cin >> to_find;
    cout << endl;

    binary_search(arr, size, to_find);
}
